-- User Mixins --

background

.background {
  @include background($success-color);
  @include size(300px, 100px);
  @include margin(null auto);
}

hide from both screenreaders and browsers

.hidden-box {
  @include size(300px, 20px);
  @include margin(null auto);
    p:hover {
    @include hidden;
  }
}

This text will hide on mouse hover

placeholders

input class="inline" [type="text"] {
  @include placeholder {
    color: $medium-gray;
  }
}
.inline {
 display: inline !important;
}// centers input box

media breakpoints

Base styles are phone first

.media-points p {
  color: red;
}

@include media(700) {
  .media-points p {
    color: blue;
  }
}

@include media(900) {
  .media-points p {
    color: rebeccapurple;
  }
}

This text will change color depending on the media breakpoint
